download standalone app

javascript code

/*
 * initialize swgg-client
 * 1. download currently-loaded apis to file swagger.json:
 *     $ curl -L "https://kaizhu256.github.io/node-swgg-github-gists/build..master..travis-ci.org/app/assets.swgg.swagger.json" > swagger.json
 * 2. npm install swgg
 *     $ npm install swgg
 * 3. run code below to initialize swgg-client
 * 4. (optional) edit file swagger.json to suit your needs
 */
var swgg;
swgg = require("swgg");
swgg.apiUpdate(require("./swagger.json"));
console.log("printing currently loaded apis ...");
console.log(JSON.stringify(Object.keys(swgg.apiDict).sort(), null, 4));
console.log("initialized swgg-client");

javascript code

/*
 * reproduce api-call "GET /gists"
 * 1. initialize swgg-client from previous step
 * 2. run code below to reproduce api-call
 */
swgg.apiDict["GET /gists"].ajax({
    "paramDict": {
        "user-agent": "swgg"
    }
}, function (error, data) {
    if (error) {
        console.error(error);
        return;
    }
    console.log(data.responseJson || data.responseText);
});
