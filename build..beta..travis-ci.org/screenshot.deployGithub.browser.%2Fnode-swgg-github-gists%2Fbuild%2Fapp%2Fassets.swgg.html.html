this zero-dependency package will provide a swagger-client for github-gists's web-apis, with a working web-demo

download standalone app

javascript code

/*
 * initialize swgg-client
 * 1. download currently-loaded apis to file swagger.json:
 *     $ curl -L "https://kaizhu256.github.io/node-swgg-github-gists/build..beta..travis-ci.org/app/assets.swgg.swagger.json" > swagger.json
 * 2. npm install swgg
 *     $ npm install swgg
 * 3. run code below to initialize swgg-client
 * 4. (optional) edit file swagger.json to suit your needs
 */
var swgg;
swgg = require("swgg");
swgg.apiUpdate(require("./swagger.json"));
console.log("printing currently loaded apis ...");
console.log(JSON.stringify(Object.keys(swgg.apiDict).sort(), null, 4));
console.log("initialized swgg-client");

 * 2. run code below to reproduce api-call

    "paramDict": {
        "user-agent": "Awesome-Octocat-App",
        "gist_id": "gist_id",
        "body": {
            "body": "Just commenting for the sake of commenting"
        }
    }

        console.error(error);
        return;
    }
